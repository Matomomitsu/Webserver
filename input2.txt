	error_page 405 /custom_405.html;
	cgi .php php-cgi;

		limit_except GET POST DELETE;

		root ./utils/uploaded;
	}

		redirect /utils/post.html;
	}
	location /delete {
		redirect /utils/delete.html;
	}

}
